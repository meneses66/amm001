Create Table CLIENT_PACKAGE
(
    ID int NOT NULL AUTO_INCREMENT PRIMARY KEY,
    CREATED datetime DEFAULT CURRENT_TIMESTAMP,
    CREATED_BY Varchar(15) NOT NULL,
    UPDATED datetime DEFAULT CURRENT_TIMESTAMP ON UPDATE CURRENT_TIMESTAMP,
    UPDATED_BY Varchar(15) NOT NULL,

    ID_CLIENT INT,
    ID_ANIMAL INT,
    ID_ORDER INT,
    ID_ORDER_ITEM INT,
    ID_PROD_SERV INT,
    PACK_QUANTITY INT,
    PACK_CONSUMED INT,
    PACK_STATUS Varchar (15),
    PACK_NAME Varchar (50),
    PACK_DATE date,
    COMMENT Varchar (200),

    constraint FK_PACKAGE_ORDER_ITEM foreign key (ID_ORDER_ITEM) references ORDER_ITEM(ID) ON DELETE CASCADE,
    INDEX INDEX_ID_CLIENT (ID_CLIENT),
    INDEX INDEX_ID_ANIMAL (ID_ANIMAL),
    INDEX INDEX_ID_ORDER (ID_ORDER)
);
