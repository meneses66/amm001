Create Table PAYMENTS
(
    ID int NOT NULL AUTO_INCREMENT PRIMARY KEY,
    CREATED datetime DEFAULT CURRENT_TIMESTAMP,
    CREATED_BY Varchar(15) NOT NULL,
    UPDATED datetime DEFAULT CURRENT_TIMESTAMP ON UPDATE CURRENT_TIMESTAMP,
    UPDATED_BY Varchar(15) NOT NULL,

    PAYMENT_TYPE Varchar(30),
    PAYMENT_DATE Date,
    PAYMENT_DESCRIPTION Varchar(50),
    PAYMENT_VALUE Decimal(10,2),
    PAYMENT_OUTPUT Decimal(10,2),
    PAYMENT_SUPPLIER Varchar(50),
    PAYMENT_CATEGORY Varchar(30),
    PAYMENT_FLAG_PROVISION BOOLEAN,
    PAYMENT_METHOD Varchar(30),


    INDEX INDEX_PAYMENT_TYPE (PAYMENT_TYPE),
    INDEX INDEX_PAYMENT_DATE (PAYMENT_DATE),
    INDEX INDEX_PAYMENT_SUPPLIER (PAYMENT_SUPPLIER),
    INDEX INDEX_PAYMENT_CATEGORY (PAYMENT_CATEGORY),
    INDEX INDEX_PAYMENT_METHOD (PAYMENT_METHOD)

);
