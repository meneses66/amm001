Create Table ORDER_ITEM
(
    ID int NOT NULL AUTO_INCREMENT PRIMARY KEY,
    CREATED datetime DEFAULT CURRENT_TIMESTAMP,
    CREATED_BY Varchar(15) NOT NULL,
    UPDATED datetime DEFAULT CURRENT_TIMESTAMP ON UPDATE CURRENT_TIMESTAMP,
    UPDATED_BY Varchar(15) NOT NULL,

    ID_ORDER INT,
    ID_PROD_SERV INT,
    QUANTITY INT,
    UNIT_VALUE Decimal(10,2),
    DISCOUNT_PERC Decimal(10,2),
    DISCOUNT_VALUE Decimal(10,2),
    VALUE_NO_DISCOUNT Decimal(10,2),
    VALUE_WITH_DISCOUNT Decimal(10,2),
    ID_PACKAGE INT,
    ID_PACKAGE_ANIMAL INT,
    PROD_SERV_TYPE Varchar (20),
    DATE Date,
    ITEM_DESCRIPTION Varchar (50),
    PRODSERV_CODE Varchar(15) NOT NULL,
    SERV_EXECUTOR Varchar (50),
    PACKAGE_AMOUNT INT,
    PACKAGE_CONSUME INT,
    PACKAGE_SEQUENCE INT,
    PACKAGE_SERVICE Varchar (50),
    PROD_SERV_CATEGORY Varchar (15),
    ID_CLIENT INT,
    BLADE Varchar (20),
    PROD_SERV_GROUP Varchar (30),
    PACKAGE_NAME Varchar (50),
    SALESPERSON Varchar (20),
    FLAG_COMISSION BOOLEAN,
    COST_CENTER Varchar (15),
    EXTERNAL_COST Decimal(10,2),
    COMISSION_PERCENTAGE Decimal(10,2),
    OLD_ID INT,
    FLAG_OTITE BOOLEAN,
    FLAG_OLHOS_VERM BOOLEAN,
    FLAG_PULGA BOOLEAN,
    FLAG_CARRAPATO BOOLEAN,
    FLAG_DERMATITE BOOLEAN,
    FLAG_FERIDA BOOLEAN,
    FLAG_OUTRO BOOLEAN,
    CHECKLIST_DESCRIPTION Varchar (1000),
    ADAP_PATA Varchar (5),
    FLAG_CONTRARIO BOOLEAN,
    TOSADOR Varchar (30),
    OI_PRICE_CASH Decimal(10,2),
    TOTAL_CASH Decimal(10,2),
    OI_PRICE_PIX Decimal(10,2),
    TOTAL_PIX Decimal(10,2),

    constraint FK_OI_ORDER foreign key (ID_ORDER) references ORDER_X(ID), 
    constraint FK_OI_PROD_SERV foreign key (ID_PROD_SERV) references PROD_SERV(ID), 
    constraint FK_OI_CLIENT foreign key (ID_CLIENT) references CLIENT(ID), 
    constraint FK_OI_PACKAGE foreign key (ID_PACKAGE) references CLIENT_PACKAGE(ID), 
    constraint FK_OI_PACKAGE_ANIMAL foreign key (ID_PACKAGE_ANIMAL) references ANIMAL(ID)

);
