Create Table SALARY
(
    ID int NOT NULL AUTO_INCREMENT PRIMARY KEY,
    CREATED datetime DEFAULT CURRENT_TIMESTAMP,
    CREATED_BY Varchar(15) NOT NULL,
    UPDATED datetime DEFAULT CURRENT_TIMESTAMP ON UPDATE CURRENT_TIMESTAMP,
    UPDATED_BY Varchar(15) NOT NULL,

    ID_EMPLOYEE	INT,
    REF_DATE Date,
    SALARY_ITEM_TYPE Varchar(30),
    SALARY_ITEM_VALUE Decimal(10,2),
    ORIGINAL_VALUE Decimal(10,2),
    POSTPONED_VALUE Decimal(10,2),
    SALARY_ITEM_STATUS Varchar(30),
    SALARY_ITEM_DESCRIPTION Varchar(100),

    INDEX INDEX_SALARY_ITEM_TYPE (SALARY_ITEM_TYPE),
    INDEX INDEX_SALARY_ITEM_STATUS (SALARY_ITEM_STATUS),
    constraint FK_SALARY_EMPLOYEE foreign key (ID_EMPLOYEE) references SUPPLIER(ID)

);
