Create Table DAILY_ACCOUNTING
(
    ID int NOT NULL AUTO_INCREMENT PRIMARY KEY,
    CREATED datetime DEFAULT CURRENT_TIMESTAMP,
    CREATED_BY Varchar(15) NOT NULL,
    UPDATED datetime DEFAULT CURRENT_TIMESTAMP ON UPDATE CURRENT_TIMESTAMP,
    UPDATED_BY Varchar(15) NOT NULL,

    DATE Date,
    INITIAL_VALUE Decimal(10,2),
    RECEIVED_CASH Decimal(10,2),
    WITHDRAW Decimal(10,2),
    FINAL_VALUE_CALC Decimal(10,2),
    FINAL_VALUE_CONFIRM Decimal(10,2),
    ADDITION Decimal(10,2),
    DIFFERENCE Decimal(10,2),
    BATHS INT,
    TOTAL_INVOICED Decimal(10,2),
    COMMENT Varchar (200),

    INDEX INDEX_DAILY_ACC_DATE (DATE)

);
