Create Table ORDER_X
(
    ID int NOT NULL AUTO_INCREMENT PRIMARY KEY,
    CREATED datetime DEFAULT CURRENT_TIMESTAMP,
    CREATED_BY Varchar(15) NOT NULL,
    UPDATED datetime DEFAULT CURRENT_TIMESTAMP ON UPDATE CURRENT_TIMESTAMP,
    UPDATED_BY Varchar(15) NOT NULL,

    ID_CLIENT INT,
    ORDER_DATE date,
    ORDER_VALUE_NO_DISCOUNT	Decimal(10,2),
    ORDER_VALUE_WITH_DISCOUNT Decimal(10,2),
    ORDER_PAID_AMOUNT Decimal(10,2),
    ORDER_DEBT Decimal(10,2),
    ORDER_VALUE_CASH Decimal(10,2),
    ORDER_VALUE_PIX Decimal(10,2),
    ORDER_OLD_ID INT,

    constraint FK_ORDER_CLIENT foreign key (ID_CLIENT) references CLIENT(ID),
    INDEX INDEX_DATE (ORDER_DATE)

);
